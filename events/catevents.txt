country_event = { # AI decreasing autonomy 
    id = ai_event.1
    title = "ai_event.EVTNAME1"
    desc = "ai_event.EVTDESC1"
    picture = LAND_MILITARY_eventPicture
    
    
    trigger = {
        ai = yes
        #is_free_or_tributary_trigger = yes
        is_at_war = no
        is_bankrupt = no
        num_of_cities = 10
        manpower_percentage = 0.2
        army_size = 15
        NOT = { overextension_percentage = 0.1 }
        NOT = { num_of_rebel_armies = 1 }
        NOT = { num_of_rebel_controlled_provinces = 1 }
        any_owned_province = {
            OR = {
                development = 10
                trade_goods = gold
            }
            controlled_by = ROOT
            is_island = no
            is_state_core = ROOT
            local_autonomy_above_min = 25
            NOT = { 
                has_province_modifier = autonomy_decreased
            }
        }
    }

    immediate = {
        random_owned_province = {
            limit = {
                OR = {
                    development = 10
                    trade_goods = gold
                }
                controlled_by = ROOT
                is_island = no
                is_state_core = ROOT
                local_autonomy_above_min = 25
                NOT = { 
                    has_province_modifier = autonomy_decreased
                }
            }
            add_province_modifier = {
                name = autonomy_decreased
                duration = 10950
            }
            add_local_autonomy = -25
        }
    }    
    option = {
    }
}
